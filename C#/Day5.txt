using System;

interface IInvoice{
    public void print(){
        
    }
}
public class HelloWorld
{
    public static void Main(string[] args)
    {
    Console.WriteLine("Enter the Invoice Type");
    int num = Convert.ToInt32(Console.ReadLine());
    
    if(num==1){
        var invoice = new clasInvoiceWithHeader();
        invoice.print();
    }
    else if(num==2){
        var invoice = new clasInvoiceWithFooter();
        invoice.print();
    }
  }
}

public class clasInvoiceWithHeader: IInvoice{
    public void print(){
        Console.WriteLine("Header Invoice");
    }
}

public class clasInvoiceWithFooter : IInvoice{
    public void print(){
       Console.WriteLine("Footer Invoice"); 
    }
}

using System;
using System.Threading.Tasks;

public class HelloWorld
{
    public static void Main(string[] args)
    {
        RunAsync();
        Console.WriteLine ("Hello Mono World");
    }
    
    public static async void RunAsync()
    {
        var res = await Run();
        Console.WriteLine(res);
    }
    
    public static async Task<string> Run(){
        var res = await Task.Run(() =>
        {
            return "Hello World";
        });
        return res;
    }
}
